    // escape HTML first
    .replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;')
    // headings
    .replace(/^### (.+)$/gm, '<div class="tag">▸ $1</div>')
    .replace(/^## (.+)$/gm, '<strong style="font-size:15px;display:block;margin:8px 0 4px;letter-spacing:.5px">$1</strong>')
    .replace(/^# (.+)$/gm,  '<strong style="font-size:16px;display:block;margin:10px 0 5px;">$1</strong>')
    // horizontal rule
    .replace(/^---$/gm, '<hr>')
    // bold + italic
    .replace(/\*\*\*(.+?)\*\*\*/g, '<strong><em>$1</em></strong>')
    .replace(/\*\*(.+?)\*\*/g, '<strong>$1</strong>')
    .replace(/\*(.+?)\*/g,   '<em>$1</em>')
    // inline code
    .replace(/`([^`]+)`/g, '<code style="background:rgba(0,255,231,0.1);padding:1px 5px;border-radius:3px;font-family:var(--mono);font-size:12px">$1</code>');
 
  // simple table
  const lines = h.split('\n');
  const out = [];
  let inTable = false, inList = false, listTag = '';
 
  for (let i = 0; i < lines.length; i++) {
    const ln = lines[i];
 
    // table
    if (ln.startsWith('|')) {
      if (!inTable) { out.push('<table>'); inTable = true; }
      const cells = ln.split('|').filter((_, idx, arr) => idx > 0 && idx < arr.length - 1);
      if (cells.every(c => /^[-: ]+$/.test(c.trim()))) continue; // separator row
      const isHeader = (i === 0 || !lines[i-1].startsWith('|')) && lines[i+1] && /^\|[-:| ]+\|$/.test(lines[i+1]);
      const tag = isHeader ? 'th' : 'td';
      out.push('<tr>' + cells.map(c => `<${tag}>${c.trim()}</${tag}>`).join('') + '</tr>');
      continue;
    }
    if (inTable) { out.push('</table>'); inTable = false; }
 
    // unordered list
    if (/^[-•]\s/.test(ln)) {
      if (!inList || listTag !== 'ul') {
        if (inList) out.push(`</${listTag}>`);
        out.push('<ul>'); inList = true; listTag = 'ul';
      }
      out.push(`<li>${ln.replace(/^[-•]\s/, '')}</li>`);
      continue;
    }
    // ordered list
    if (/^\d+\.\s/.test(ln)) {
      if (!inList || listTag !== 'ol') {
        if (inList) out.push(`</${listTag}>`);
        out.push('<ol>'); inList = true; listTag = 'ol';
      }
      out.push(`<li>${ln.replace(/^\d+\.\s/, '')}</li>`);
      continue;
    }
    if (inList) { out.push(`</${listTag}>`); inList = false; listTag = ''; }
 
    if (ln.trim() === '') { out.push('<br>'); continue; }
    out.push(ln);
  }
  if (inTable) out.push('</table>');
  if (inList)  out.push(`</${listTag}>`);
 
  return out.join('\n');
}

async function callClaude(userMessage) {
  history.push({ role: 'user', content: userMessage });
 
  const body = {
    model: 'claude-sonnet-4-20250514',
    max_tokens: 1000,
    system: SYSTEM_PROMPT,
    messages: history,
  };
 
  const res = await fetch('https://api.anthropic.com/v1/messages', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(body),
  });
 
  if (!res.ok) {
    const err = await res.json().catch(() => ({}));
    throw new Error(err?.error?.message || `HTTP ${res.status}`);
  }
 
  const data = await res.json();
  const reply = data.content?.find(b => b.type === 'text')?.text || '';
  history.push({ role: 'assistant', content: reply });
  return reply;
}

// RESPOND
// ─────────────────────────────────────────────────────────────────────────────
async function respond(query) {
  if (busy || !query.trim()) return;
  addUserMsg(query);
  setUI(true);
  const typing = addTypingIndicator();
 
  try {
    const reply = await callClaude(query);
    removeTypingIndicator();
    addBotMsg(renderMarkdown(reply));
  } catch (e) {
    removeTypingIndicator();
    addBotMsg(`<em style="color:rgba(255,100,100,0.8)">Sorry, something went wrong: ${e.message}.<br>Please email <strong>danieldavidwilds@gmail.com</strong> or call <strong>(604) 901-1692</strong> directly.</em>`);
    // remove last user message from history so user can retry
    history.pop();
  } finally {
    setUI(false);
    $('tin').focus();
  }
}

function send() {
  const tin = $('tin');
  const q = tin.value.trim();
  if (!q || busy) return;
  tin.value = '';
  respond(q);
}
